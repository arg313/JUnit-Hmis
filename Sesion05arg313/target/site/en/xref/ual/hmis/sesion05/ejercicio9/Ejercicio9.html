7   	public ArrayList<String> listaCadenasOrdenadas(
8   			final ArrayList<String> cadena1,
9   			final ArrayList<String>  cadena2) {
10  		ArrayList<String> resultado = new ArrayList<String>();
11  
12  		for (int i = 0; i < cadena1.size(); i++) {
13  			for (int j = 0; j < cadena2.size(); j++) {
14  				if (cadena1.get(i).equals(cadena2.get(j))
15  						&& !resultado.contains(cadena1.get(i))) {
16  						resultado.add(cadena1.get(i));
17  				}
18  			}
19  		}
20  
21  		resultado.sort(new ComparadorCadenas());
22  		return resultado;
23  	}
